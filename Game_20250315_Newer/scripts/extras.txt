function finishSetup() {
    const rollLogPanel = document.getElementById("rollLogPanel");

    rollLogPanel.innerHTML += `<p>Your character is ready for the adventure!</p>`;


    // Add a Next button
    rollLogPanel.innerHTML += `<button class="game-button" onclick="goToNextRoom()">Next</button>`;
}

// Function to add roll results to the secondary roll log panel
function addToRollLog(rollResult) {
    const rollLogPanel = document.getElementById('rollLogPanel');
    const rollMessage = document.createElement('p');
    rollMessage.textContent = rollResult;
    rollLogPanel.appendChild(rollMessage);
    
    // Auto scroll to the bottom of the roll log
    rollLogPanel.scrollTop = rollLogPanel.scrollHeight;
}

function goToNextRoom() {
    // Check if it's the first time (no active room yet)
    if (latestActiveRoom === null) {
        // First time, go to showRoomDetails and set it as the latest active room
        showRoomDetails();
        latestActiveRoom = "showRoomDetails";
    } else {
        // Go to the last active room
        window[latestActiveRoom]();
    }
}
